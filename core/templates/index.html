collab-workspace.tsx 

import { useCollab } from '@collab/react';
import { useState, useEffect } from 'react';

export default function CollabWorkspace() {
  const [project, setProject] = useState(null);
  const { collaborators, chatMessages } = useCollab('user/repo');

  useEffect(() => {
    // Initialize the collaborative workspace
    initializeCollabSession(collaborators);
  }, [collaborators]);

  return (
    <div className="collab-workspace">
      <CodeEditor />
      <ChatPanel messages={chatMessages} />
      <ActivityFeed />
      <CollaboratorsList users={collaborators} />
    </div>
  );
}

  collaborators,
  activities,
  threads
} = useCollab('user/repo');

return (
  <CollabWorkspace>
    <CodeEditor />
    <ActivityFeed data={activities} />
    <CollaboratorsList users={collaborators} />
    <DiscussionThreads threads={threads} />
  </CollabWorkspace>
);
}
